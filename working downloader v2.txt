#!/usr/bin/env python3

import sys, requests, time, os, threading, queue
from http.server import HTTPServer, BaseHTTPRequestHandler
from urllib.parse import parse_qs, urlparse
from PyQt6.QtWidgets import (
    QApplication, QWidget, QVBoxLayout, QHBoxLayout, QLineEdit, 
    QPushButton, QTableWidget, QTableWidgetItem, QHeaderView, QLabel
)
from PyQt6.QtCore import QThread, pyqtSignal, QTimer, Qt

HOME = os.path.expanduser("~")

file_types = {
    "Videos":     ["mp4", "mkv", "avi", "mov", "webm"],
    "Documents":  ["pdf", "doc", "docx", "txt", "ppt", "pptx"],
    "Compressed": ["zip", "rar", "7z", "tar", "gz"],
    "Programs":   ["exe", "bin", "appimage", "deb", "rpm"]
}

def choose_folder(filename):
    ext = filename.split(".")[-1].lower() if "." in filename else ""
    for folder, extensions in file_types.items():
        if ext in extensions:
            path = os.path.join(HOME, "Downloads", folder)
            os.makedirs(path, exist_ok=True)
            return path
    path = os.path.join(HOME, "Downloads", "Others")
    os.makedirs(path, exist_ok=True)
    return path

def get_filename_from_url(url):
    parsed = urlparse(url)
    filename = os.path.basename(parsed.path)
    return filename if (filename and "." in filename) else "downloaded_file"

url_queue = queue.Queue()

class BridgeHandler(BaseHTTPRequestHandler):
    def do_GET(self):
        try:
            q = parse_qs(urlparse(self.path).query)
            url = q.get("url", [None])[0]
            filename = q.get("filename", [None])[0]
            if url:
                url_queue.put((url, filename))
                self.send_response(200); self.end_headers()
                self.wfile.write(b"OK")
        except Exception:
            self.send_response(500); self.end_headers()

    def log_message(self, *args): pass

def start_bridge_server(port=9999):
    server = HTTPServer(("127.0.0.1", port), BridgeHandler)
    threading.Thread(target=server.serve_forever, daemon=True).start()

class DownloadThread(QThread):
    progress = pyqtSignal(int)
    speed    = pyqtSignal(str)
    finished = pyqtSignal(str)

    def __init__(self, url, filename):
        super().__init__()
        self.url = url
        self.filename = filename
        self.running = True

    def run(self):
        try:
            with requests.get(self.url, stream=True, allow_redirects=True, timeout=15, verify=False) as r:
                r.raise_for_status()
                total = int(r.headers.get("content-length", 0))
                folder = choose_folder(self.filename)
                filepath = os.path.join(folder, self.filename)
                
                downloaded, start_time = 0, time.time()
                with open(filepath, "wb") as f:
                    for chunk in r.iter_content(chunk_size=8192):
                        if not self.running:
                            self.finished.emit("Cancelled"); return
                        if chunk:
                            f.write(chunk); downloaded += len(chunk)
                            if total > 0: self.progress.emit(int(downloaded * 100 / total))
                            elapsed = time.time() - start_time
                            if elapsed > 0: self.speed.emit(f"{downloaded / elapsed / 1024:.2f} KB/s")
            self.finished.emit("Finished")
        except Exception as e:
            self.finished.emit(f"Error: {str(e)}")

    def stop(self): self.running = False

class DownloadManager(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Linux Download Manager")
        self.resize(700, 500)
        layout = QVBoxLayout()

        self.url_input = QLineEdit()
        self.url_input.setPlaceholderText("Link will appear here automatically...")
        layout.addWidget(self.url_input)

        # Square Button Layout
        btn_layout = QHBoxLayout()
        btn_layout.setSpacing(10)
        
        # Base style for square buttons
        base_style = "font-weight: bold; color: white; border-radius: 5px; font-size: 14px; min-width: 120px; min-height: 80px;"

        self.start_btn = QPushButton("Start\nDownload")
        self.start_btn.setStyleSheet(f"{base_style} background-color: #27ae60;")
        self.start_btn.clicked.connect(self.start_manual)

        self.cancel_btn = QPushButton("Cancel\nDownload")
        self.cancel_btn.setStyleSheet(f"{base_style} background-color: #c0392b;")
        self.cancel_btn.clicked.connect(self.cancel_last)

        self.clear_btn = QPushButton("Clear\nList")
        self.clear_btn.setStyleSheet(f"{base_style} background-color: #f39c12;")
        self.clear_btn.clicked.connect(self.clear_list)

        btn_layout.addWidget(self.start_btn)
        btn_layout.addWidget(self.cancel_btn)
        btn_layout.addWidget(self.clear_btn)
        btn_layout.addStretch() # Pushes buttons to the left
        
        layout.addLayout(btn_layout)

        self.status_label = QLabel("Status: Monitoring Firefox Downloads...")
        layout.addWidget(self.status_label)

        self.table = QTableWidget(); self.table.setColumnCount(4)
        self.table.setHorizontalHeaderLabels(["File Name", "Progress", "Speed", "Status"])
        self.table.horizontalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Stretch)
        layout.addWidget(self.table)

        self.setLayout(layout); self.threads = []
        start_bridge_server()
        self.timer = QTimer(); self.timer.timeout.connect(self.check_queue); self.timer.start(500)

    def check_queue(self):
        while not url_queue.empty():
            url, filename = url_queue.get()
            if not filename or filename == "None":
                filename = get_filename_from_url(url)
            self.url_input.setText(url)
            self._enqueue(url, filename)

    def start_manual(self):
        url = self.url_input.text().strip()
        if url: self._enqueue(url, get_filename_from_url(url))

    def _enqueue(self, url, filename):
        row = self.table.rowCount(); self.table.insertRow(row)
        self.table.setItem(row, 0, QTableWidgetItem(filename))
        self.table.setItem(row, 1, QTableWidgetItem("0%"))
        self.table.setItem(row, 2, QTableWidgetItem("0 KB/s"))
        self.table.setItem(row, 3, QTableWidgetItem("Downloading"))

        thread = DownloadThread(url, filename)
        self.threads.append(thread)
        thread.progress.connect(lambda v, r=row: self.table.setItem(r, 1, QTableWidgetItem(f"{v}%")))
        thread.speed.connect(lambda s, r=row: self.table.setItem(r, 2, QTableWidgetItem(s)))
        thread.finished.connect(lambda m, r=row: self.table.setItem(r, 3, QTableWidgetItem(m)))
        thread.start()

    def cancel_last(self):
        # Stops the most recently added download that is still running
        for thread in reversed(self.threads):
            if thread.isRunning():
                thread.stop()
                self.status_label.setText("Action: Cancelled most recent download.")
                break

    def clear_list(self):
        self.table.setRowCount(0)
        self.threads = [t for t in self.threads if t.isRunning()]
        self.status_label.setText("Action: List cleared.")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = DownloadManager(); window.show()
    sys.exit(app.exec())
